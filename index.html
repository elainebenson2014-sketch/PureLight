PROFILES (one row per signed-in user) ---------- create table if not exists pl_profiles ( id uuid primary key references auth.users(id) on delete cascade, full_name text not null default '', email text not null default '', role text not null default 'student' check (role in ('student','instructor')), created_at timestamptz not null default now() ); -- Auto-create a profile whenever someone signs up create or replace function pl_handle_new_user() returns trigger language plpgsql security definer set search_path = public as $$ begin insert into pl_profiles (id, full_name, email) values (new.id, coalesce(new.raw_user_meta_data->>'full_name',''), new.email) on conflict (id) do nothing; return new; end; $$; drop trigger if exists pl_on_auth_user_created on auth.users; create trigger pl_on_auth_user_created after insert on auth.users for each row execute function pl_handle_new_user(); -- Helper used by security policies create or replace function pl_is_instructor() returns boolean language sql security definer set search_path = public as $$ select exists (select 1 from pl_profiles where id = auth.uid() and role = 'instructor'); $$; -- ---------- 2. BOOKS ---------- create table if not exists pl_books ( id uuid primary key default gen_random_uuid(), title text not null, author text not null default '', description text not null default '', pages int not null default 0, file_path text, -- path inside the 'purelight' storage bucket created_by uuid references pl_profiles(id) on delete set null, created_at timestamptz not null default now() ); -- ---------- 3. TESTS ---------- create table if not exists pl_tests ( id uuid primary key default gen_random_uuid(), book_id uuid references pl_books(id) on delete set null, title text not null, description text not null default '', file_path text, -- optional uploaded test PDF created_by uuid references pl_profiles(id) on delete set null, created_at timestamptz not null default now() ); -- ---------- 4. QUESTIONS ---------- create table if not exists pl_questions ( id uuid primary key default gen_random_uuid(), test_id uuid not null references pl_tests(id) on delete cascade, position int not null default 0, type text not null check (type in ('mc','tf','short','essay')), prompt text not null default '', points int not null default 0, options jsonb not null default '[]'::jsonb, -- for multiple choice correct_answer text -- mc: option index as text; tf: 'true'/'false' ); -- ---------- 5. SUBMISSIONS ---------- create table if not exists pl_submissions ( id uuid primary key default gen_random_uuid(), test_id uuid not null references pl_tests(id) on delete cascade, student_id uuid not null references pl_profiles(id) on delete cascade, answers jsonb not null default '{}'::jsonb, -- { question_id: response } manual jsonb not null default '{}'::jsonb, -- { question_id: awarded_points } status text not null default 'submitted' check (status in ('submitted','graded')), score int, max_score int, feedback text not null default '', submitted_at timestamptz not null default now() ); -- ---------- 6. MESSAGES (email + announcements) ---------- create table if not exists pl_messages ( id uuid primary key default gen_random_uuid(), sender_id uuid references pl_profiles(id) on delete set null, sender_name text not null default '', recipient text not null default 'all', -- 'all' or a pl_profiles.id subject text not null default '', body text not null default '', created_at timestamptz not null default now() );
